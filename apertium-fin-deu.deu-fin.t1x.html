comparison-mangler

Parametres: 1

      
    if (tl[1]['a_adj'] ≟ <num.ord>) then:
       let $comparison ≔ "";
  
    elseif (sl[1]['a_comp'] ≟ <comp>) then:
       let $comparison ≔ <com>;
  
    elseif (sl[1]['a_comp'] ≟ <sup>) then:
       let $comparison ≔ <sup>;
  
    else:
       let $comparison ≔ <pos>;

Parametres: 1

    else:
       let $voice ≔ <actv>;

number-mangler

Parametres: 1

      
    if (
              tl[1]['a_number'] ≟ <sp>
              tl[1]['a_case'] ≟ <nom>
              tl[1]['lem'] ≟ "kaikki"
            ) then:
       let $number ≔ <sp>;
  
    elseif (
              tl[1]['a_number'] ≟ <sp>
              tl[1]['a_case'] ≟ <gen>
              tl[1]['lem'] ≟ "mikä"
            ) then:
       let $number ≔ <sp>;
  
    elseif (tl[1]['lem'] ≟ "koko") then:
       let $number ≔ "";
  
    elseif (
              $tense ≟ <pri>
              tl[1]['a_conneg'] ≟ <conneg>
            ) then:
       let $number ≔ "";
  
    elseif (tl[1]['a_nominal'] ≟ <n.compound-only-L>) then:
       let $number ≔ "";
  
    elseif (tl[1]['a_number'] ≟ <sg>) then:
       let $number ≔ <sg>;
  
    elseif (tl[1]['a_number'] ≟ <pl>) then:
       let $number ≔ <pl>;
  
    else:
       let $number ≔ <sg>;
